    function labuladuo(s, l, r) {
        while (l >= 0 && r <= s.length && s.charAt(l) === s.charAt(r)) {
            l--;
            r++
        }
        return s.substring(l + 1, r)
    }

    let res = ""
    for (let i = 0; i < s.length; i++) {
