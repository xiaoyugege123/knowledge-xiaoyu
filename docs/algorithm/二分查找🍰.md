```javascript
var searchInsert = function (nums, target) {
  let left = 0,
    right = nums.length - 1;
  while (left <= right) {
    const mid = Math.floor((left + right) / 2);
    if (nums[mid] == target) {
      return mid;
    } else if (nums[mid] > target) {
      right = mid - 1;
    } else {
      left = mid + 1;
    }
  }
  return left; //left就是安插的那个点！！！
```

## 33. 搜索旋转排序数组
```js
/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number}
 */
var search = function (nums, target) {
    //这里的条件判断！！！
    const n = nums.length
    if (n == 0) return -1
    if (n == 1) return nums[0] == target ? 0 : -1

    
    let left = 0, right = nums.length - 1
    while (left <= right) {
        const mid = Math.floor((left + right) / 2)
        if (nums[mid] == target) {
            return mid
        }
        // 题目中整数数组 nums 按升序排列，数组中的值 互不相同 。
        if (nums[mid] >= nums[0]) {//这一段是连续的！！
            if (nums[left] <= target && target < nums[mid]) {
                right = mid - 1
            } else {
                left = mid + 1
            }
        } else {
            if (nums[mid] < target && target <= nums[right]) {
                left = mid + 1
            } else {
                right = mid - 1
            }
        }
    }
    return -1
};
```

```javascript
var findMin = function (nums) {
    let low = 0, high = nums.length - 1
    while (low < high) {
        const pivot = Math.floor((low + high) / 2)
        if (nums[pivot] < nums[high]) {
            high = pivot
        } else {
            low = pivot + 1
        }
    }
    return nums[low]
};
```

## 69. x 的平方根

```javascript
/**
 * @param {number} x
 * @return {number}
 */
var mySqrt = function (x) {
  //用二分法进行求解：左闭右开区间
  let low = 0,
    high = Math.ceil(x / 2); //优化方案点
  while (low < high) {
    const mid = Math.ceil((high + low) / 2);
    const res = mid * mid;
    if (res == x) {
      return mid;
    } else if (res > x) {
      high = mid - 1;
    } else {
      low = mid;
    }
  }
  return low;
};
```

## 367. 有效的完全平方数

```javascript
/**
 * @param {number} num
 * @return {boolean}
 */
var isPerfectSquare = function (num) {
  //4=1+3 9=1+3+5 16=1+3+5+7以此类推，模仿它可以使用一个while循环，
  // 不断减去一个从1开始不断增大的奇数，若最终减成了0，说明是完全平方数，否则，不是。
  let num1 = 1;
  while (num > 0) {
    num -= num1;
    num1 += 2;
  }
  return num == 0;
};
```

## 4. 寻找两个正序数组的中位数

其实这个题我没有用二分，自己模拟了一下

```js
/**
 * @param {number[]} nums1
 * @param {number[]} nums2
 * @return {number}
 */
var findMedianSortedArrays = function (nums1, nums2) {
  // 这个题目蛮简单的，听思路：while+条件判断遍历到中间位置！
  const len = nums1.length + nums2.length;
  let centerIndex;
  if (len % 2 === 0) {
    // 指向后面一个
    centerIndex = len / 2;
  } else {
    centerIndex = Math.floor(len / 2);
  }
  let preVal, curVal;
  let n1 = 0,
    n2 = 0; // 分别是num1 & num2 的索引
  let count = -1;
  while (count !== centerIndex) {
    // 需要记录下最后指向的值和最后停止的上一次值
    preVal = curVal;
    count++;
    if (n1 == nums1.length) {
      curVal = nums2[n2];
      n2++;
      continue;
    }
    if (n2 == nums2.length) {
      curVal = nums1[n1];
      n1++;
      continue;
    }
    if (nums1[n1] > nums2[n2]) {
      curVal = nums2[n2];
      n2++;
    } else {
      curVal = nums1[n1];
      n1++;
    }
  }
  // 判断奇偶
  if (len % 2 === 0) {
    return (preVal + curVal) / 2;
  } else {
    return curVal;
  }
};
```

## 295. 数据流的中位数

```js
var MedianFinder = function () {
  this.arr = [];
};

/**
 * @param {number} num
 * @return {void}
 */
MedianFinder.prototype.addNum = function (num) {
  // 数组的插入，用二分法进行插入
  let left = 0;
  let right = this.arr.length - 1;

  while (left <= right) {
    let mid = Math.floor((left + right) / 2);

    if (this.arr[mid] === num) {
      this.arr.splice(mid, 0, num);
      return;
    } else if (this.arr[mid] < num) {
      left = mid + 1;
    } else {
      right = mid - 1;
    }
  }
  this.arr.splice(left, 0, num);
};

/**
 * @return {number}
 */
MedianFinder.prototype.findMedian = function () {
  const len = this.arr.length;
  let centerIndex;
  if (len % 2 === 0) {
    centerIndex = len / 2;
    return (this.arr[centerIndex - 1] + this.arr[centerIndex]) / 2;
  } else {
    centerIndex = Math.floor(len / 2);
    return this.arr[centerIndex];
  }
};

/**
 * Your MedianFinder object will be instantiated and called as such:
 * var obj = new MedianFinder()
 * obj.addNum(num)
 * var param_2 = obj.findMedian()
 */
```

```js
function insertInOrder(arr, num) {
  let left = 0;
  let right = arr.length - 1;

  while (left <= right) {
    let mid = Math.floor((left + right) / 2);

    if (arr[mid] === num) {
      arr.splice(mid, 0, num);
      return arr;
    } else if (arr[mid] < num) {
      left = mid + 1;
    } else {
      right = mid - 1;
    }
  }

  arr.splice(left, 0, num);
  return arr;
}
```
