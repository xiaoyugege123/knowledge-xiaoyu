var diagonalSort = function (mat) {

    //å…ˆè¿›è¡Œæ”¶é›†ç®¡é“æ•°æ®

    }
    //è¿›è¡Œæ’åº

    //è¿›è¡Œå¡«å……

        }
    }
    return mat;
};

function getOnes(board, i, j) {
    const arr = [
        board[i]?.[j + 1],
        board[i]?.[j - 1],
        board[i + 1]?.[j],
        board[i + 1]?.[j - 1],
        board[i + 1]?.[j + 1],
        board[i - 1]?.[j],
        board[i - 1]?.[j + 1],
        board[i - 1]?.[j - 1]
    ]
