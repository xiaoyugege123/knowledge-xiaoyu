 * class ListNode {
 *     val: number
 *     next: ListNode | null

function addTwoNumbers(l1: ListNode | null, l2: ListNode | null): ListNode | null {
    let p1 = l1, p2 = l2;
    const dummy = new ListNode();
    let p = dummy
    let carry;
    while (p1 != null || p2 != null) {
        let res = 0;
        if (p1) {
            res += p1.val
            p1 = p1.next
        }
        if (p2) {
            res += p2.val
            p2 = p2.next
        };
        if (carry) {
            res += carry;
        }
        carry = res / 10 | 0

        const node = new ListNode(res % 10)
        p.next = node;
        p = p.next;
    }
    if (carry) {
        const node = new ListNode(carry)
        p.next = node
    }
    return dummy.next
};
